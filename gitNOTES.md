#Check current changes:
git status

#Add all files to next commit:
git add .

#See branches:
git branch

#New branch:
git checkout -b [new name of branch]

#Switch to another existing branch:
git checkout [branch name]

#Create a commit:
git commit -m "ad comment here"

#Push commit to git:
git push -u origin [name of branch]

#Drop changes:
git stash drop
